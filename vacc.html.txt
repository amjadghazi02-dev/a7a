function save() { localStorage.setItem(STORAGE_KEY, JSON.stringify(animals)); }
function uid()  { return Date.now().toString(36) + Math.random().toString(36).slice(2); }
function today(){ return new Date().toISOString().split("T")[0]; }
function daysUntil(d) {
  if (!d) return null;
  return Math.round((new Date(d) - new Date(today())) / 86400000);
}

// ═══════════════════════ VIEWS ═══════════════════════
function showView(view, id) {
  currentView = view;
  if (id) selectedId = id;
  const hdr = document.getElementById("headerTitle");
  const back = document.getElementById("btnBack");
  const add  = document.getElementById("btnAdd");

  back.style.display = view !== "list" ? "block" : "none";
  add.style.display  = view === "list" ? "block" : "none";

  if (view === "list")       { hdr.textContent = "دفتر المطاعيم"; renderList(); }
  if (view === "add")        { hdr.textContent = "تسجيل حيوان جديد"; renderAddAnimal(); }
  if (view === "detail")     { hdr.textContent = "ملف الحيوان"; renderDetail(); }
  if (view === "addVaccine") { hdr.textContent = "إضافة تطعيم"; renderAddVaccine(); }
}

function goBack() {
  if (currentView === "addVaccine") showView("detail");
  else { selectedId = null; showView("list"); }
}

// ═══════════════════════ LIST ═══════════════════════
let searchVal = "";
function renderList() {
  const c = document.getElementById("content");
  const td = today();
  const filtered = animals.filter(a =>
    !searchVal ||
    a.name.includes(searchVal) ||
    (a.ownerName||"").includes(searchVal) ||
    a.species.includes(searchVal)
  );

  if (animals.length === 0) {
    c.innerHTML = `
      <div class="empty">
        <div class="big">🐾</div>
        <h3>لا يوجد سجلات بعد</h3>
        <p>اضغط «+ إضافة» لتسجيل حيوان جديد</p>
      </div>`;
    return;
  }

  let cards = filtered.map(a => {
    const vaccCount = (a.vaccines||[]).length;
    let statusTag = "";
    (a.vaccines||[]).forEach(v => {
      const d = daysUntil(v.nextDate);
      if (d !== null && d < 0)  statusTag = `<span class="tag-overdue">موعد متأخر!</span>`;
      else if (d !== null && d <= 7 && !statusTag) statusTag = `<span class="tag-soon">قريباً</span>`;
    });
    return `
    <div class="animal-card" onclick="showView('detail','${a.id}')">
      <div class="ac-emoji">${SPECIES[a.species]||"🐾"}</div>
      <div class="ac-body">
        <div class="ac-name">${esc(a.name)}</div>
        <div class="ac-sub">${esc(a.species)}${a.breed?" · "+esc(a.breed):""}</div>
        <div class="ac-owner">${a.ownerName?"👤 "+esc(a.ownerName):""}</div>
        <div class="ac-vacc">💉 ${vaccCount} تطعيم ${statusTag}</div>
      </div>
      <div class="ac-arrow">›</div>
    </div>`;
  }).join("");

  c.innerHTML = `
    <div class="search-wrap">
      <span class="search-icon">🔍</span>
      <input type="search" placeholder="ابحث باسم الحيوان أو المالك..." value="${esc(searchVal)}"
        oninput="searchVal=this.value;renderList()" />
    </div>
    ${filtered.length === 0 ? `<div class="empty"><div class="big">🔍</div><h3>لا نتائج</h3><p>جرب بحث آخر</p></div>` : `<div class="animals">${cards}</div>`}
  `;
}

function saveVaccine() {
const sel = document.getElementById(“v_sel”).value;
const name = sel === “**custom**”
? document.getElementById(“v_custom”).value.trim()
: sel;
const date = document.getElementById(“v_date”).value;
if (!name) { toast(“⚠️ يرجى اختيار أو إدخال اسم التطعيم”); return; }
if (!date) { toast(“⚠️ يرجى إدخال تاريخ التطعيم”); return; }

const a = animals.find(x => x.id === selectedId);
if (!a) return;
a.vaccines = a.vaccines || [];
a.vaccines.unshift({
id: uid(), name, date,
nextDate: document.getElementById(“v_next”).value,
batchNo:  document.getElementById(“v_batch”).value.trim(),
vetName:  document.getElementById(“v_vet”).value.trim(),
notes:    document.getElementById(“v_notes”).value.trim()
});
save();
toast(“✅ تم حفظ التطعيم”);
showView(“detail”);
}

// ═══════════════════════ HELPERS ═══════════════════════
function esc(s) {
return String(s||””).replace(/&/g,”&”).replace(/</g,”<”).replace(/>/g,”>”).replace(/”/g,”"”);
}

function toast(msg) {
const t = document.getElementById(“toast”);
t.textContent = msg; t.classList.add(“show”);
clearTimeout(t._t);
t._t = setTimeout(() => t.classList.remove(“show”), 2500);
}

function showConfirm(title, sub, cb) {
document.getElementById(“confirmTitle”).textContent = title;
document.getElementById(“confirmSub”).textContent   = sub;
confirmCallback = cb;
document.getElementById(“confirmOverlay”).classList.add(“open”);
}
function closeConfirm() { document.getElementById(“confirmOverlay”).classList.remove(“open”); }
document.getElementById(“confirmOk”).onclick = () => { closeConfirm(); if(confirmCallback) confirmCallback(); };
document.getElementById(“confirmOverlay”).addEventListener(“click”, e => { if(e.target === e.currentTarget) closeConfirm(); });
